function printBillingReport(){
  var ws=fmtD(bWkStart);var we=fmtD(bWkEnd());
  var pending=[];
  for(var i=0;i<D.jobs.length;i++){
    var j=D.jobs[i];var billed=getBilledToDate(j.JobID);
    if(j.NoBill==='Yes')continue;
    if(j.EndDate){var je=String(j.EndDate).split('T')[0];if(je<=we&&billed<100)pending.push({job:j,type:'Complete',pct:100-billed,billed:billed});}
    if(j.PartialBilling==='Yes'&&j.PartialBillDate&&j.BillingPaused!=='Yes'){var pbd=String(j.PartialBillDate).split('T')[0];if(pbd<=we&&billed<100)pending.push({job:j,type:'Partial',pct:parseFloat(j.PartialPercent)||0,billed:billed});}
    if(j.BillingPaused==='Yes'&&j.PartialBilling==='Yes'&&billed<100)pending.push({job:j,type:'Partial',pct:parseFloat(j.PartialPercent)||0,billed:billed,paused:true});
  }
  var seen={};var dedP=[];for(var i=0;i<pending.length;i++){var k=String(pending[i].job.JobID);if(!seen[k]){seen[k]=true;dedP.push(pending[i]);}}pending=dedP;
  var confirmed=[];
  if(D.billingLog){for(var i=0;i<D.billingLog.length;i++){var lg=D.billingLog[i];var ld=String(lg.Date).split('T')[0];if(ld>=ws&&ld<=we){var j2=null;for(var ji=0;ji<D.jobs.length;ji++){if(String(D.jobs[ji].JobID)===String(lg.JobID)){j2=D.jobs[ji];break;}}confirmed.push({log:lg,job:j2});}}}
  var confirmedBill=[];var confirmedNB=[];
  for(var i=0;i<confirmed.length;i++){if(confirmed[i].job&&confirmed[i].job.NoBill==='Yes')confirmedNB.push(confirmed[i]);else confirmedBill.push(confirmed[i]);}
  var b='<h2>Billing Report</h2><div class="sub">'+bFmtWk()+'</div>';
  if(pending.length){
    b+='<h3 style="margin:16px 0 6px;font-size:13px;color:#555">Pending ('+pending.length+')</h3>';
    b+='<table><thead><tr><th>Job</th><th>Type</th><th>This Cycle</th><th>Billed</th><th>New Total</th><th>Contract</th><th>Notes</th></tr></thead><tbody>';
    for(var i=0;i<pending.length;i++){
      var it=pending[i];var j=it.job;var nt=Math.min(it.billed+it.pct,100);
      b+='<tr><td><strong>'+j.JobNum+' - '+j.JobName+'</strong>'+(isPW(j)?' <span class="chip" style="background:#6d28d9;color:#fff">PW</span>':'')+(it.paused?' <span class="chip" style="background:#e65100;color:#fff">PAUSED</span>':'')+'</td>';
      b+='<td>'+it.type+'</td><td>'+it.pct+'%</td><td>'+it.billed+'%</td><td><strong>'+nt+'%</strong></td>';
      b+='<td>'+(j.Amount?'$'+j.Amount:'—')+'</td><td>'+(j.BillingNotes||'—')+'</td></tr>';
    }
    b+='</tbody></table>';
  }
  if(confirmedBill.length){
    b+='<h3 style="margin:16px 0 6px;font-size:13px;color:#555">Confirmed This Week ('+confirmedBill.length+')</h3>';
    b+='<table><thead><tr><th>Job</th><th>Type</th><th>%</th><th>Cumulative</th><th>Date</th></tr></thead><tbody>';
    for(var i=0;i<confirmedBill.length;i++){
      var c=confirmedBill[i];var lg=c.log;var j=c.job;
      b+='<tr><td>'+(j?j.JobNum+' - '+j.JobName:'Job '+lg.JobID)+(j&&isPW(j)?' <span class="chip" style="background:#6d28d9;color:#fff">PW</span>':'')+'</td>';
      b+='<td>'+lg.Type+'</td><td>'+lg.Percent+'%</td><td>'+lg.CumulativePercent+'%</td><td>'+lg.Date+'</td></tr>';
    }
    b+='</tbody></table>';

  printWin('Billing Report - '+bFmtWk(),b);

function doImportCSV(jid){
  var ta=document.getElementById('csv-'+jid);if(!ta)return;
  var csv=ta.value;if(!csv.trim()){msg('Paste CSV first','wrn');return;}
  msg('Extracting...','inf');sSync('ing');

    sSync('ok');

    else{msg(r.message||'No materials found','err');}
  }).withFailureHandler(function(e){msg('Error: '+(e?e.message:'unknown'),'err');sSync('bad');}).importMaterials(String(jid),csv);

function pickSOW(jid){
  var inp=document.createElement('input');
  inp.type='file';
  inp.accept='.xlsx,.xls,.csv';
  inp.onchange=function(e){
    var file=e.target.files[0];
    if(!file)return;
    msg('Reading '+file.name+'...','inf');
    var reader=new FileReader();
    reader.onload=function(ev){
      try{
        var data=new Uint8Array(ev.target.result);
        var wb=XLSX.read(data,{type:'array',cellDates:true});
        var ws=wb.Sheets[wb.SheetNames[0]];
        var rows=XLSX.utils.sheet_to_json(ws,{header:1,defval:''});
        var mats=parseSOW(rows);
        if(!mats.length){msg('No materials found in file','err');return;}
        msg('Found '+mats.length+' materials, importing...','inf');
        sSync('ing');
        google.script.run.withSuccessHandler(function(r){
          sSync('ok');
          if(r&&r.success){msg(r.count+' materials imported','ok');refreshData();}
          else{msg((r&&r.message)||'Import failed','err');}
        }).withFailureHandler(function(e){
          msg('Error: '+(e?e.message:'unknown'),'err');sSync('bad');
        }).importSOWMaterials(String(jid),mats);
      }catch(ex){msg('Error reading file: '+ex.message,'err');}
    };
    reader.readAsArrayBuffer(file);
  };
  inp.click();
}

function parseSOW(rows){
  var results=[];
  var phase='';var task='';var inMat=false;
  for(var i=0;i<rows.length;i++){
    var r=rows[i];
    var c0=String(r[0]||'').trim();
    var c1=String(r[1]||'').trim();
    var c2=String(r[2]||'').trim();
    if(c0.match(/^Day\s*\d/i)||c0==='Daily Average Required'){
      phase=c0;task=c2;inMat=false;continue;
    }
    if(c1==='SOW'||c1==='Task'){
      if(c1==='SOW')task=c2;
      continue;
    }
    if(c0==='Material'&&(c2==='Kit Size'||c2.toLowerCase().indexOf('kit')>=0)){
      inMat=true;continue;
    }
    if(c0==='Crew Count'||c0.indexOf('WTC')===0){
      inMat=false;continue;
    }
    if(inMat&&c0&&c0!=='None'&&c0!=='undefined'){
      results.push({
        Phase:phase||'General',
        Task:(task||'').substring(0,200),
        Material:c0,
        KitSize:c2||'',
        QtyOrdered:String(r[3]||''),
        Mils:String(r[4]||''),
        CoverageRate:String(r[5]||''),
        MixTime:String(r[6]||''),
        MixSpeed:String(r[7]||''),
        MatStatus:'Not Ordered',
        ArrivalDate:'',
        Notes:''
      });
    }
  }
  return results;
}

function clearMats(jid){
  if(!confirm('Clear all materials for this job?'))return;
  sSync('ing');
  google.script.run.withSuccessHandler(function(){
    var nm=[];for(var i=0;i<D.materials.length;i++){if(String(D.materials[i].JobID)!==String(jid))nm.push(D.materials[i]);}D.materials=nm;
    sSync('ok');msg('Materials cleared','wrn');rMaterials();
  }).withFailureHandler(function(){msg('Error','err');sSync('bad');}).deleteMaterials(String(jid));
}

}
